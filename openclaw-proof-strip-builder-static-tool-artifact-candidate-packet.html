    .strip { display: grid; gap: 8px; border: 1px solid #d8dee9; padding: 12px; }

<body>
  <main>
    <h1>Proof Strip Builder</h1>
    <section class="layout" aria-label="Builder workspace">
      <div>
        <label>Buyer profile <textarea id="buyer"></textarea></label>
        <label>Public result note <textarea id="result"></textarea></label>
        <label>Proof type <select id="proofType"><option>Public review excerpt</option><option>Before and after note</option><option>Workflow sample</option></select></label>
        <button type="button" id="saveDraft">Save local draft</button>
        <button type="button" id="clearDraft">Clear local draft</button>
      </div>
      <output id="preview" class="strip" aria-live="polite">Enter owner-reviewed public facts.</output>
    </section>
    <textarea id="manualExport" readonly aria-label="Manual export"></textarea>
  </main>
  <script>
    const stateKey = 'openclaw_proof_strip_builder_draft_v1';
    const buyer = document.querySelector('#buyer');
    const result = document.querySelector('#result');
    const proofType = document.querySelector('#proofType');
    const preview = document.querySelector('#preview');
    const manualExport = document.querySelector('#manualExport');
    function clean(value) { return String(value || '').replace(/[<>]/g, '').trim(); }
    function render() {
      const data = { buyer: clean(buyer.value), result: clean(result.value), proofType: proofType.value };
      preview.textContent = [data.proofType, data.buyer, data.result, 'Owner review required before public use.'].filter(Boolean).join(' | ');
      manualExport.value = JSON.stringify(data, null, 2);
    }
    document.querySelector('#saveDraft').addEventListener('click', () => { localStorage.setItem(stateKey, manualExport.value); });
    document.querySelector('#clearDraft').addEventListener('click', () => { localStorage.removeItem(stateKey); buyer.value = ''; result.value = ''; render(); });
    [buyer, result, proofType].forEach((node) => node.addEventListener('input', render));
    render();
  </script>
</body>
